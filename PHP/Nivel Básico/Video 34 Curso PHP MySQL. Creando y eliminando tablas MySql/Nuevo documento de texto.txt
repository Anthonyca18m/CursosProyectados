CREAR Y ELIMINAR TABLAS EN UNA BBDD MYSQL
	ESTRUCTURA DE UNA TABLA
		CREATE TABLE NOMBRE_TABLA(
		
		ID_VARIABLE INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
		VARIABLE_NOMBRE VARCHAR(255) NOT NULL
		//.....
		);

	DROPEAR O ELIMINAR UNA TABLA	
	
		DELETE TABLE NOMBRE_TABLA;
		DROP TABLE NOMBRE_TABLA;
		DELETE FROM NOMBRE_TABLA;//PARA ELIMINAR DATOS DE UNA TABLA

MOSTRAR CAMPOS DE TABLA
		SELECT * FROM NOMBRE_TABLA; //MOSTRAR TODOS LOS ATRIBUTOS DE LA TABLA
